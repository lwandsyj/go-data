字符串回文原理： 字符串反转以后和原来字符串一致

  反转字符串： 因为字符串是不可变的，因此可以先把字符转成rune(使用rune 是因为rune 支持中文) 切片，然后使用双指针

        //Reverse 反转
        func Reverse(s string) string {
            if s == "" {
                return s
            }
            result := []rune(s)
            l := len(result)
            begin := 0
            end := l - 1
            for begin < end {
                result[begin], result[end] = result[end], result[begin]
                begin++
                end--
            }
            return string(result)
        }
